# User
{
    email: {
        type: String,
        required: true,
        unique: true,
    },
    password: {
        type: String,
        required: true,
    },
    firstName: {
        type: String,
        required: true,
    },
    lastName: {
        type: String,
    },
    phone: {
        type: Number,
        required: true,
    },
    birth: {
        type: Date,
        required: true,
    },
    status: {
        type: String,
        enum: ['active', 'pending'],
        default: 'pending',
    }
}
